## Task 3: Auto-generate JteClassRegistry at build time

**Context:** `JteClassRegistry` has 36 entries but 41 `.kte` templates exist. 5 templates are missing and would fail in production mode. The registry must be manually updated in 3 places (file + metadata + registry) when adding templates. Auto-generation eliminates this class of bug.

**Files:**
- Create: `scripts/generate-jte-class-registry.ps1`
- Modify: `platform-web/src/main/kotlin/io/github/rygel/outerstellar/platform/infra/JteClassRegistry.kt` (regenerated by script)
- Modify: `platform-web/pom.xml` (add exec-maven-plugin to run script before compile)

**Design decisions:**
- The script scans `src/main/jte/**/*.kte` files and generates the Kotlin source for `JteClassRegistry.kt`
- Categories (page/fragment/component/layout) are determined by directory and naming conventions:
  - `layouts/` → `layoutClasses`
  - `components/` → `componentClasses`
  - Direct under `web/`, name contains `Fragment` or `Form` → `fragmentClasses`
  - Everything else → `pageClasses`
- The script outputs the complete file content, which is written to the source tree
- A Maven `exec-maven-plugin` invocation in the `generate-sources` phase runs the script before compilation

- [ ] **Step 1: Create the generation script**

Create `scripts/generate-jte-class-registry.ps1`:

```powershell
<#
.SYNOPSIS
Generates JteClassRegistry.kt from actual .kte template files.

.DESCRIPTION
Scans src/main/jte/ for .kte files and generates the JteClassRegistry object
with correct imports and class references. Run before compilation to keep the
registry in sync with templates.

.EXAMPLE
pwsh scripts/generate-jte-class-registry.ps1
#>

param(
    [string]$JteSourceDir = "platform-web/src/main/jte",
    [string]$OutputFile = "platform-web/src/main/kotlin/io/github/rygel/outerstellar/platform/infra/JteClassRegistry.kt",
    [string]$BasePackage = "gg.jte.generated.precompiled.outerstellar"
)

$ErrorActionPreference = "Stop"

# Collect all .kte files
$templates = Get-ChildItem -LiteralPath $JteSourceDir -Recurse -Filter "*.kte" -File

if ($templates.Count -eq 0) {
    Write-Error "No .kte templates found in $JteSourceDir"
    exit 1
}

class TemplateEntry {
    [string]$RelativePath   # e.g. "io/github/rygel/outerstellar/platform/web/HomePage.kte"
    [string]$FQCN           # e.g. "gg.jte.generated.precompiled.outerstellar.io.github.rygel.outerstellar.platform.web.JteHomePageGenerated"
    [string]$SimpleClass    # e.g. "JteHomePageGenerated"
    [string]$Category       # page, fragment, component, layout
    [string]$Import         # full import line
}

$entries = [System.Collections.Generic.List[TemplateEntry]]::new()

foreach ($t in $templates) {
    $relPath = $t.FullName.Substring((Resolve-Path $JteSourceDir).Path.Length + 1).Replace('\', '/')

    # Directory portion as package suffix
    $dir = [System.IO.Path]::GetDirectoryName($relPath).Replace('/', '.').Replace('\', '.')

    # Sanitize template name: remove .kte, strip hyphens and dots
    $baseName = [System.IO.Path]::GetFileNameWithoutExtension($relPath)
    $sanitized = $baseName -replace '[-\.]', ''

    $simpleClass = "Jte${sanitized}Generated"

    if ($dir -and $dir -ne '.') {
        $fqcn = "${BasePackage}.${dir}.${simpleClass}"
    } else {
        $fqcn = "${BasePackage}.${simpleClass}"
    }

    # Categorize
    $category = "page"
    if ($relPath -match '/layouts/') {
        $category = "layout"
    } elseif ($relPath -match '/components/') {
        $category = "component"
    } elseif ($baseName -match '(Fragment|Form)$') {
        $category = "fragment"
    }

    $entry = [TemplateEntry]::new()
    $entry.RelativePath = $relPath
    $entry.FQCN = $fqcn
    $entry.SimpleClass = $simpleClass
    $entry.Category = $category
    $entry.Import = "import ${fqcn}"
    $entries.Add($entry)
}

# Group by category
$pages = $entries | Where-Object { $_.Category -eq "page" } | Sort-Object SimpleClass
$fragments = $entries | Where-Object { $_.Category -eq "fragment" } | Sort-Object SimpleClass
$components = $entries | Where-Object { $_.Category -eq "component" } | Sort-Object SimpleClass
$layouts = $entries | Where-Object { $_.Category -eq "layout" } | Sort-Object SimpleClass
$allSorted = $entries | Sort-Object FQCN

# Build output
$sb = [System.Text.StringBuilder]::new()

[void]$sb.AppendLine("package io.github.rygel.outerstellar.platform.infra")
[void]$sb.AppendLine("")

# Imports
foreach ($e in $allSorted) {
    [void]$sb.AppendLine($e.Import)
}
[void]$sb.AppendLine("import org.slf4j.LoggerFactory")
[void]$sb.AppendLine("")

# Class
[void]$sb.AppendLine("object JteClassRegistry {")
[void]$sb.AppendLine("    private val logger = LoggerFactory.getLogger(JteClassRegistry::class.java)")
[void]$sb.AppendLine("")

# pageClasses
[void]$sb.AppendLine("    private val pageClasses = listOf(")
foreach ($e in $pages) {
    [void]$sb.AppendLine("        $($e.SimpleClass)::class.java,")
}
[void]$sb.AppendLine("    )")
[void]$sb.AppendLine("")

# fragmentClasses
[void]$sb.AppendLine("    private val fragmentClasses = listOf(")
foreach ($e in $fragments) {
    [void]$sb.AppendLine("        $($e.SimpleClass)::class.java,")
}
[void]$sb.AppendLine("    )")
[void]$sb.AppendLine("")

# componentClasses
[void]$sb.AppendLine("    private val componentClasses = listOf(")
foreach ($e in $components) {
    [void]$sb.AppendLine("        $($e.SimpleClass)::class.java,")
}
[void]$sb.AppendLine("    )")
[void]$sb.AppendLine("")

# layoutClasses
[void]$sb.AppendLine("    private val layoutClasses = listOf(")
foreach ($e in $layouts) {
    [void]$sb.AppendLine("        $($e.SimpleClass)::class.java,")
}
[void]$sb.AppendLine("    )")
[void]$sb.AppendLine("")

# allClasses
[void]$sb.AppendLine("    val allClasses: List<Class<*>> = pageClasses + fragmentClasses + componentClasses + layoutClasses")
[void]$sb.AppendLine("")

# classMap + init + getTemplateClass (preserve existing logic)
[void]$sb.AppendLine("    private val classMap: Map<String, Class<*>> = allClasses.associateBy { it.name }")
[void]$sb.AppendLine("")
[void]$sb.AppendLine("    init {")
[void]$sb.AppendLine("        allClasses.forEach { clazz ->")
[void]$sb.AppendLine("            try {")
[void]$sb.AppendLine("                val loaded = Class.forName(clazz.name)")
[void]$sb.AppendLine("                logger.debug(\"Loaded template class: {}\", loaded.name)")
[void]$sb.AppendLine("            } catch (e: ClassNotFoundException) {")
[void]$sb.AppendLine("                logger.warn(\"Template class not found: {}\", clazz.name)")
[void]$sb.AppendLine("            }")
[void]$sb.AppendLine("        }")
[void]$sb.AppendLine("    }")
[void]$sb.AppendLine("")
[void]$sb.AppendLine("    fun getTemplateClass(templateName: String): Class<*>? {")
[void]$sb.AppendLine("        val packageSuffix = templateName")
[void]$sb.AppendLine("            .replace(\".kte\", \"\")")
[void]$sb.AppendLine("            .replace(\"/\", \".\")")
[void]$sb.AppendLine("            .replace(\"-\", \"\")")
[void]$sb.AppendLine("            .replace(\".\", \"\")")
[void]$sb.AppendLine("        val className = \"${BasePackage}.\${templateName.replace(\"/\", \".\").replace(\".kte\", \"\").split(\"/\").last()}\" +")
[void]$sb.AppendLine("            \"Generated\"")
[void]$sb.AppendLine("        return classMap.entries.firstOrNull { it.key.contains(packageSuffix, ignoreCase = true) }?.value")
[void]$sb.AppendLine("    }")
[void]$sb.AppendLine("}")

$sb.ToString() | Set-Content -LiteralPath $OutputFile -Encoding UTF8

Write-Host "Generated JteClassRegistry.kt with $($entries.Count) template entries"
Write-Host "  Pages: $($pages.Count), Fragments: $($fragments.Count), Components: $($components.Count), Layouts: $($layouts.Count)"
```
